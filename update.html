// npm install express body-parser twilio // server.js const express = require('express'); const bodyParser = require('body-parser'); const twilio = require('twilio'); const app = express(); app.use(bodyParser.json()); const accountSid = 'YOUR_TWILIO_ACCOUNT_SID'; const authToken = 'YOUR_TWILIO_AUTH_TOKEN'; const client = twilio(accountSid, authToken); let generatedOtp = null; app.post('/send-otp', (req, res) => { const { username } = req.body; // Generate OTP generatedOtp = Math.floor(100000 + Math.random() * 900000).toString(); // Send OTP to user's phone (assuming username is linked to phone number) client.messages.create({ body: `Your OTP is ${generatedOtp}`, from: 'YOUR_TWILIO_PHONE_NUMBER', to: 'USER_PHONE_NUMBER' // Replace with the actual user's phone number }).then(message => { res.json({ success: true }); }).catch(error => { res.json({ success: false, error: error.message }); }); }); app.post('/verify-otp', (req, res) => { const { otp } = req.body; if (otp === generatedOtp) { res.json({ success: true }); } else { res.json({ success: false }); } }); app.listen(3000, () => { console.log('Server is running on port 3000'); });
